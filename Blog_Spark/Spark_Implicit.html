spark.implicits._ 

    导入的隐式转换的 implicits._ 不是spark包里面的，而是在SparkSession在bulider()或者new
SQLContext(SparkContext(conf))所产生的对象，因此需要修改一下对象，便可调用spark的toDS()或
者toDF()方法了
    import org.apache.spark.{SparkConf, SparkContext}
import org.apache.spark.sql.{DataFrame, Dataset, SQLContext, SparkSession}

object CreateDf {
  def main(args: Array[String]): Unit = {
    val conf = new SparkConf().setAppName("test").setMaster("local[*]")
    val spark1 = SparkSession.builder().config(conf).getOrCreate()
    import spark1.implicits._
    //    val sc = SparkContext(conf)
    //    val spark2 = new SQLContext(sc)
    //    import spark2.implicits._
    val ds: Dataset[Int] = Seq(2, 1).toDS()
    val ds2: DataFrame = Seq(2, 1).toDF()
  }
}
